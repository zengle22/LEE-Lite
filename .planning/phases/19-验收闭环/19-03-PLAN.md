"""spec_adapter — SPEC_ADAPTER_COMPAT bridge layer.

Truth source: ADR-054 §2.2 SPEC_ADAPTER_COMPAT format specification.
Maps api-test-spec/*.md and e2e-journey-spec/*.md to TESTSET-compatible YAML.

This module provides:
- ApiSpecAdapter: parses API test specs, outputs SPEC_ADAPTER_COMPAT
- E2eSpecAdapter: parses E2E journey specs, outputs SPEC_ADAPTER_COMPAT
- spec_to_testset(): main conversion function

import re

from pathlib import Path
from typing import Any

@dataclass
class SpecAdapterConfig:
    source_chain: str  # "api" | "spec_e2e"

@dataclass

    unit_ref: str
    title: str
    trigger_action: str = ""

    preconditions: list[str] = field(default_factory=list)

class ApiSpecAdapter:
    """Adapter for API test specs (api-test-spec/*.md) → SPEC_ADAPTER_COMPAT.

    Maps fields per ADR-054 §2.2.2:
    - case_id → unit_ref
    - capability + scenario_type → title
    - endpoint.method + endpoint.path → trigger_action
    - assertions → pass_conditions / fail_conditions
    - source_feat_ref → acceptance_ref
    """

    def __init__(self, spec_data: dict[str, Any]):
        self._data = spec_data

    @property
    def coverage_id(self) -> str:
        return self._data.get("coverage_id", self._data.get("case_id", ""))

    @property
    def case_id(self) -> str:
        return self._data.get("case_id", "")

    @property
    def capability(self) -> str:
        return self._data.get("capability", "")

    @property
    def scenario_type(self) -> str:
        return self._data.get("scenario_type", "main")

    @property
    def priority(self) -> str:
        return self._data.get("priority", "P1")

    @property
    def source_feat_ref(self) -> str:
        return self._data.get("source_feat_ref", "")

    @property
    def endpoint(self) -> str:
        ep = self._data.get("endpoint", "")
        if isinstance(ep, dict):
            method = ep.get("method", "GET")
            path = ep.get("path", "/")
            return f"{method} {path}"
        return str(ep)

    @property
    def pass_conditions(self) -> list[str]:
        assertions = self._data.get("assertions", [])
        return [a for a in assertions if not a.startswith("!")]

    @property
    def fail_conditions(self) -> list[str]:
        assertions = self._data.get("assertions", [])
        return [a.lstrip("!") for a in assertions if a.startswith("!")]

    @property
    def preconditions(self) -> list[str]:
        return self._data.get("preconditions", [])

    @property
    def request_body(self) -> dict[str, Any]:
        req = self._data.get("request", {})
        if isinstance(req, dict):
            return req.get("body", {})
        return {}

    def to_unit(self) -> dict[str, Any]:
        """Convert to TESTSET unit dict with _source_coverage_id."""
        unit = {
            "unit_ref": self.case_id,
            "title": f"{self.capability}: {self.scenario_type}",
            "trigger_action": self.endpoint,
            "pass_conditions": self.pass_conditions,
            "fail_conditions": self.fail_conditions,
            "preconditions": self.preconditions,
            "test_data": self.request_body,
            "acceptance_ref": self.source_feat_ref,
            "priority": self.priority,
            "required_evidence": self._data.get("evidence_required", []),
            "_source_coverage_id": self.coverage_id,
            "_api_extension": {
                "scenario_type": self.scenario_type,
                "capability": self.capability,
                "source_feat_ref": self.source_feat_ref,
            },
        }
        return unit

class E2eSpecAdapter:
    """Adapter for E2E journey specs (e2e-journey-spec/*.md) → SPEC_ADAPTER_COMPAT.

    Maps fields per ADR-054 §2.2.3:
    - spec_id → unit_ref
    - title → title
    - priority → priority
    - entry_point → page_path
    - user_steps → ui_steps
    - expected_ui_states → pass_conditions
    - anti_false_pass_checks → fail_conditions
    """

    def __init__(self, spec_data: dict[str, Any]):
        self._data = spec_data

    @property
    def spec_id(self) -> str:
        return self._data.get("spec_id", "")

    @property
    def coverage_id_ref(self) -> str:
        return self._data.get("coverage_id_ref", "")

    @property
    def title(self) -> str:
        return self._data.get("title", "")

    @property
    def priority(self) -> str:
        return self._data.get("priority", "P1")

    @property
    def entry_point(self) -> str:
        return self._data.get("entry_point", "")

    @property
    def scenario_type(self) -> str:
        return self._data.get("scenario_type", "main")

    @property
    def user_steps(self) -> list[dict[str, Any]]:
        return self._data.get("user_steps", [])

    @property
    def expected_ui_states(self) -> list[str]:
        return self._data.get("expected_ui_states", [])

    @property
    def anti_false_pass_checks(self) -> list[str]:
        return self._data.get("anti_false_pass_checks", [])

    def _resolve_selector(self, step: dict[str, Any]) -> str | None:
        """Resolve selector from step per ADR-054 §5.1 R-2."""
        target = step.get("target", "")
        target_format = step.get("target_format", "css_selector")
        if target_format == "css_selector":
            return target
        elif target_format == "semantic":
            action = step.get("action", "")
            return f"[data-action={action}]"
        elif target_format == "text":
            return f"text={target}"
        return target if target else None

    def to_unit(self) -> dict[str, Any]:
        """Convert to TESTSET unit dict with _source_coverage_id + _e2e_extension."""
        ui_steps = []
        for step in self.user_steps:
            action = step.get("action", "")
            selector = self._resolve_selector(step)
            ui_steps.append(action)
            if selector:
                ui_steps.append(f"[{selector}]")

        ui_step_metadata = []
        for step in self.user_steps:
            ui_step_metadata.append({
                "action": step.get("action", ""),
                "type": step.get("type", ""),
                "target": step.get("target", ""),
                "target_format": step.get("target_format", "css_selector"),
                "timeout": step.get("timeout"),
            })

        unit = {
            "unit_ref": self.spec_id,
            "title": self.title,
            "page_path": self.entry_point,
            "ui_steps": ui_steps,
            "pass_conditions": self.expected_ui_states,
            "fail_conditions": self.anti_false_pass_checks,
            "priority": self.priority,
            "_source_coverage_id": self.coverage_id_ref,
            "_e2e_extension": {
                "ui_step_metadata": ui_step_metadata,
                "scenario_type": self.scenario_type,
                "expected_persistence": self._data.get("expected_persistence", []),
            },
        }
        return unit

def spec_to_testset(
    spec_items: list[dict[str, Any]],
    chain: str,
    feat_ref: str | None = None,
    proto_ref: str | None = None,
) -> dict[str, Any]:
    """Convert spec items to SPEC_ADAPTER_COMPAT format.

    Args:
        spec_items: List of spec dicts (from parsing api-test-spec or e2e-journey-spec)
        chain: "api" or "e2e"
        feat_ref: Feature reference (API chain)
        proto_ref: Prototype reference (E2E chain)

    Returns:
        SPEC_ADAPTER_COMPAT dict with test_units.
    """
    if chain == "api":
        units = [ApiSpecAdapter(item).to_unit() for item in spec_items]
    else:
        units = [E2eSpecAdapter(item).to_unit() for item in spec_items]

    result = {
        "ssot_type": "SPEC_ADAPTER_COMPAT",
        "test_set_id": f"spec-adapter-{feat_ref or proto_ref or 'unknown'}",
        "execution_modality": "api" if chain == "api" else "web_e2e",
        "source_chain": "api" if chain == "api" else "spec_e2e",
        "test_units": units,
    }

    if feat_ref:
        result["feat_ref"] = feat_ref
    if proto_ref:
        result["prototype_ref"] = proto_ref

    return result

"""environment_provision — generates TEST_ENVIRONMENT_SPEC files.

Truth source: ADR-054 §2.3 environment_provision module specification.
Generates ssot/environments/ENV-{id}.yaml from feat.environment_assumptions + user parameters.

This module provides:
- EnvProvisionInput: dataclass for input parameters
- provision_environment(): generates ENV YAML file
"""

from __future__ import annotations

from dataclasses import dataclass, field
from datetime import datetime, timezone
from pathlib import Path
from typing import Any

import yaml


@dataclass
class EnvProvisionInput:
    """Input parameters for environment provision.

    Per ADR-054 §2.3.1 Input Contract:
    - User parameters override feat environment_assumptions
    - modality: "api" | "web_e2e" | "cli"
    """
    feat_ref: str | None = None
    proto_ref: str | None = None
    base_url: str = "http://localhost:8000"
    modality: str = "api"
    browser: str = "chromium"
    timeout: int = 30000
    feat_assumptions: list[str] = field(default_factory=list)
    # Extended per R-3: --app-url / --api-url support
    app_url: str | None = None
    api_url: str | None = None


@dataclass
class EnvConfig:
    """Output dataclass from environment provision."""
    env_id: str
    test_environment_ref: str
    execution_modality: str
    base_url: str
    app_base_url: str | None = None
    api_base_url: str | None = None
    browser: str = "chromium"
    timeout: int = 30000
    required_fields: list[str] = field(default_factory=list)
    feature_flags: dict[str, Any] = field(default_factory=dict)
    test_data_config: dict[str, Any] = field(default_factory=dict)
    source_feat_ref: str | None = None
    source_proto_ref: str | None = None
    generated_by: str = "environment_provision"
    generated_at: str = ""


def provision_environment(
    input_params: EnvProvisionInput,
    output_dir: str | Path | None = None,
) -> EnvConfig:
    """Generate TEST_ENVIRONMENT_SPEC from input parameters.

    Args:
        input_params: EnvProvisionInput with feat_ref/proto_ref, base_url, modality
        output_dir: Directory for output ENV file. Defaults to ssot/environments/

    Returns:
        EnvConfig dataclass with all environment fields.

    Per ADR-054 §2.3.2 Output Specification.
    Per R-3: supports --app-url and --api-url for separated architecture.
    """
    if output_dir is None:
        output_dir = Path("ssot/environments")
    else:
        output_dir = Path(output_dir)

    # Determine env_id
    env_id = f"ENV-{input_params.feat_ref or input_params.proto_ref or 'DEFAULT'}"

    # Build base URLs per R-3
    if input_params.modality == "web_e2e":
        app_base_url = input_params.app_url or input_params.base_url
        api_base_url = input_params.api_url
    else:
        app_base_url = None
        api_base_url = input_params.api_url or input_params.base_url

    # Build required_fields based on modality
    if input_params.modality == "api":
        required_fields = ["base_url"]
    elif input_params.modality == "web_e2e":
        required_fields = ["app_base_url", "browser"]
        if api_base_url:
            required_fields.append("api_base_url")
    else:
        required_fields = []

    config = EnvConfig(
        env_id=env_id,
        test_environment_ref=str(output_dir / f"{env_id}.yaml"),
        execution_modality=input_params.modality,
        base_url=input_params.base_url,
        app_base_url=app_base_url,
        api_base_url=api_base_url,
        browser=input_params.browser,
        timeout=input_params.timeout,
        required_fields=required_fields,
        source_feat_ref=input_params.feat_ref,
        source_proto_ref=input_params.proto_ref,
        generated_by="environment_provision",
        generated_at=datetime.now(timezone.utc).isoformat(),
    )

    return config


def write_env_config(config: EnvConfig, output_path: str | Path | None = None) -> None:
    """Write EnvConfig to YAML file.

    Args:
        config: EnvConfig from provision_environment()
        output_path: Destination path. Defaults to config.test_environment_ref
    """
    if output_path is None:
        output_path = config.test_environment_ref

    p = Path(output_path)
    p.parent.mkdir(parents=True, exist_ok=True)

    env_dict = {
        "ssot_type": "TEST_ENVIRONMENT_SPEC",
        "test_environment_ref": config.test_environment_ref,
        "execution_modality": config.execution_modality,
        "base_url": config.base_url,
        "browser": config.browser,
        "timeout": config.timeout,
        "required_fields": config.required_fields,
        "feature_flags": config.feature_flags,
        "test_data_config": config.test_data_config,
        "source_feat_ref": config.source_feat_ref,
        "source_proto_ref": config.source_proto_ref,
        "generated_by": config.generated_by,
        "generated_at": config.generated_at,
    }

    # Add optional fields if present
    if config.app_base_url:
        env_dict["app_base_url"] = config.app_base_url
    if config.api_base_url:
        env_dict["api_base_url"] = config.api_base_url

    with open(p, "w", encoding="utf-8") as f:
        yaml.safe_dump(env_dict, f, default_flow_style=False, sort_keys=False)
