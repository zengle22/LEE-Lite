```bash
#!/usr/bin/env bash
set -euo pipefail
SCRIPT_DIR="$(cd "$(dirname "$0")" && pwd)"
SPEC_PATH=""
TEST_DIR=""
MANIFEST_PATH=""
OUTPUT_DIR=""
WORKSPACE="${PWD}"
BASE_URL=""

while [[ $# -gt 0 ]]; do
  case "$1" in
    --spec-path) SPEC_PATH="$2"; shift 2;;
    --test-dir) TEST_DIR="$2"; shift 2;;
    --manifest-path) MANIFEST_PATH="$2"; shift 2;;
    --output-dir) OUTPUT_DIR="$2"; shift 2;;
    --workspace) WORKSPACE="$2"; shift 2;;
    --base-url) BASE_URL="$2"; shift 2;;
    *) shift;;
  esac
done

# Validate inputs
bash "${SCRIPT_DIR}/validate_input.sh" "${SPEC_PATH}" "${TEST_DIR}" "${MANIFEST_PATH}"

RUN_ID="run-$(date +%s)-$$"
EVIDENCE_DIR="${OUTPUT_DIR:-${WORKSPACE}/ssot/tests/.artifacts/evidence/${RUN_ID}}"
mkdir -p "${EVIDENCE_DIR}"

# Execute via Python module
python -c "
import sys; sys.path.insert(0, '${WORKSPACE}/cli/lib')
from api_test_exec import run_api_test_exec
result = run_api_test_exec(
    spec_path='${SPEC_PATH}',
    test_dir='${TEST_DIR}',
    manifest_path='${MANIFEST_PATH}',
    evidence_dir='${EVIDENCE_DIR}',
    run_id='${RUN_ID}',
    base_url='${BASE_URL}',
)
import json; print(json.dumps(result))
"

# Validate outputs
bash "${SCRIPT_DIR}/validate_output.sh" "${EVIDENCE_DIR}"
```

Export:
```python
def run_api_test_exec(
    spec_path: str,
    test_dir: str,
    manifest_path: str,
    evidence_dir: str,
    run_id: str,
    base_url: str = "",
) -> dict:
```

Implementation steps:
1. Load spec YAML to get case_id, coverage_id, evidence_required list
2. Set EVIDENCE_DIR environment variable to evidence_dir for generated tests to use
3. Set BASE_URL environment variable if provided
4. Run pytest via subprocess:
```python
import subprocess
result = subprocess.run(
    ["pytest", test_dir, "--junitxml", f"{evidence_dir}/results.xml",
     "-v", "--tb=short"],
    capture_output=True, text=True,
    env={**os.environ, "EVIDENCE_DIR": evidence_dir, "EVIDENCE_RUN_ID": run_id, "BASE_URL": base_url},
)
```
5. Parse junitxml to get per-case results (use xml.etree.ElementTree):
   - For each testcase element, extract name and check for failure/error child elements
   - Map to pass/fail status
6. For each test case:
   a. Look for evidence file at {evidence_dir}/{coverage_id}.evidence.yaml
   b. If exists: validate with validate_evidence() from qa_schemas
   c. Check all spec.evidence_required items are present in evidence dict
   d. Build result dict: {coverage_id, passed, evidence_path}
7. Call backfill_manifest(manifest_path, results, run_id)
8. Return summary dict with total, passed, failed, error counts, evidence_dir, manifest_path

Use pathlib.Path for all paths. Handle subprocess failures gracefully (pytest returns non-zero on test failures — this is expected, not an error).
