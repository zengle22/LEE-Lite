**1. Enhance _format_frz_list() — FIXED columns with '-' for empty values:**

Per review consensus: Use FIXED columns with '-' for empty values instead of conditional column rendering. This prevents breaking downstream scripts that parse by column position.

```python
def _format_frz_list(frz_records: list[dict[str, Any]]) -> str:
    if not frz_records:
        return "No FRZ packages registered"

    sorted_records = sorted(...)

    # FIXED header — always shows all columns, '-' for empty values
    header = ["FRZ_ID", "STATUS", "REV_TYPE", "PREV_FRZ", "CREATED_AT", "MSC_VALID"]

    # Build rows with '-' for empty values
    rows = []
    for rec in sorted_records:
        frz_id = rec.get("frz_id", "")
        status = rec.get("status", "")
        rev_type = rec.get("revision_type", "new")
        prev_frz = rec.get("previous_frz_ref", "-")  # '-' instead of empty
        created_at = rec.get("created_at", "")
        msc_valid = "yes" if rec.get("msc_valid") else "no"
        rows.append([frz_id, status, rev_type, prev_frz, created_at, msc_valid])

    # Format as aligned table
    col_widths = [len(h) for h in header]
    for row in rows:
        for i, cell in enumerate(row):
            col_widths[i] = max(col_widths[i], len(str(cell)))

    header_line = "  ".join(h.ljust(col_widths[i]) for i, h in enumerate(header))
    separator = "  ".join("-" * col_widths[i] for i in range(len(header)))
    result_lines = [header_line, separator]
    for row in rows:
        result_lines.append("  ".join(str(cell).ljust(col_widths[i]) for i, cell in enumerate(row)))

    return "\n".join(result_lines)
```

**2. Add circular revision chain prevention in freeze_frz():**

Before calling register_frz() with revision_type="revise", check that creating this revision would not form a circular chain.

```python
def _check_circular_revision(frz_registry: dict, new_frz_id: str, previous_frz: str) -> list[str]:
    """Check if adding new_frz_id -> previous_frz would create a circular chain.
    
    Returns list of FRZ IDs in the chain if circular, empty list if safe.
    """
    visited = set()
    current = previous_frz
    while current:
        if current == new_frz_id:
            # Circular chain detected: new_frz_id -> ... -> current -> new_frz_id
            return list(visited) + [current]
        if current in visited:
            # Already visited but not new_frz_id — broken chain, not circular
            break
        visited.add(current)
        # Get the previous_frz_ref of the current FRZ
        rec = frz_registry.get(current)
        if rec:
            current = rec.get("previous_frz_ref")
        else:
            current = None
    return []
```
