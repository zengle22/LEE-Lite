```python
def derive_grade(change_class: str) -> GradeLevel:
    """Derive grade level from change_class deterministically.
    
    Fail-safe default: if change_class is missing/invalid, return MAJOR with warning.
    This is safer than MINOR — an unknown classification should escalate to human review.
    """
    import warnings
    grade = CHANGE_CLASS_TO_GRADE.get(change_class)
    if grade is None:
        warnings.warn(
            f"Unknown change_class '{change_class}', defaulting to MAJOR for safety. "
            "Add this value to CHANGE_CLASS_TO_GRADE in patch_schema.py."
        )
        return GradeLevel.MAJOR
    return grade
```

**classify_change() function — per review consensus:**
- Input: text string, optional paths list for fallback
- Scan ALL indicator lists, collect ALL matching dimensions
- Apply "semantic dominates" rule: if any semantic indicator matches -> GradeLevel.MAJOR regardless of other matches
- Determine confidence:
  - "high" = single dimension matched
  - "medium" = multiple same-grade dimensions matched (e.g., visual + interaction, both MINOR)
  - "low" = conflicting dimensions or no indicators matched
- Set needs_human_review = True if confidence is "low" OR if conflicting dimensions detected
- FALLBACK: if keyword matching returns NO indicators, import and call `_suggest_change_class` from patch_auto_register.py as fallback. This contradicts the old "Do NOT import" instruction — review consensus says to use file-pattern classification as fallback.

```python
def classify_change(text: str, paths: list[str] | None = None) -> dict:
    """Classify change with dimensions_detected, confidence, needs_human_review."""
    lower = text.lower()
    matched_dimensions: list[str] = []
    
    for dimension, indicators in CLASSIFICATION_RULES.items():
        if any(ind in lower for ind in indicators):
            # Map dimension name to change_class
            if "semantic" in dimension:
                matched_dimensions.append("semantic")
            elif "interaction" in dimension:
                matched_dimensions.append("interaction")
            elif "visual" in dimension:
                matched_dimensions.append("visual")
    
    if not matched_dimensions:
        # FALLBACK: use file-pattern classification from patch_auto_register.py
        from cli.lib.patch_auto_register import _suggest_change_class
        if paths:
            fallback_class = _suggest_change_class(paths)
            matched_dimensions.append(fallback_class.value)
        else:
            matched_dimensions.append("other")
    
    # Semantic dominates: if semantic is in matched dimensions -> MAJOR
    has_semantic = "semantic" in matched_dimensions
    if has_semantic:
        change_class = ChangeClass.semantic
        grade_level = GradeLevel.MAJOR
    elif len(matched_dimensions) == 1:
        # Single dimension matched
        dim = matched_dimensions[0]
        change_class = ChangeClass(dim) if dim in [e.value for e in ChangeClass] else ChangeClass.other
        grade_level = derive_grade(change_class.value)
    else:
        # Multiple dimensions, no semantic -> take highest-grade (all MINOR here)
        # Prefer interaction over visual for specificity
        if "interaction" in matched_dimensions:
            change_class = ChangeClass.interaction
        elif "visual" in matched_dimensions:
            change_class = ChangeClass.visual
        else:
            change_class = ChangeClass.other
        grade_level = derive_grade(change_class.value)
    
    # Confidence calculation
    if len(matched_dimensions) == 1:
        confidence = "high"
    elif len(matched_dimensions) > 1 and not has_semantic:
        confidence = "medium"  # multiple same-grade dimensions
    elif has_semantic and len(matched_dimensions) > 1:
        confidence = "medium"  # semantic dominates but mixed input
    else:
        confidence = "low"
    
    needs_human_review = confidence == "low"
    
    return {
        "change_class": change_class.value,
        "grade_level": grade_level.value,
        "dimensions_detected": matched_dimensions,
        "confidence": confidence,
        "needs_human_review": needs_human_review,
    }
```
