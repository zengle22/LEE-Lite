def _load_patch_yaml(path: Path) -> dict[str, Any] | None:
    try:
        with open(path, "r", encoding="utf-8") as f:
            data = yaml.safe_load(f)
        if not isinstance(data, dict):
            return None
        # Derive grade_level if not present, using derive_grade from patch_schema
        if "grade_level" not in data and "change_class" in data:
            data["grade_level"] = derive_grade(data["change_class"]).value
            data["grade_derived_from"] = "patch_schema.derive_grade"
        return data
    except (OSError, yaml.YAMLError):
        return None
```

def test_derive_grade_consistency_across_callers():
    """Assert that _load_patch_yaml, summarize_patch_for_context, and patch_aware_context.py's
    summarize_patch all produce the SAME grade for the same change_class."""
    from cli.lib.patch_schema import derive_grade
    from cli.lib.patch_context_injector import summarize_patch_for_context, _load_patch_yaml
    import sys; sys.path.insert(0, 'skills/ll-patch-aware-context/scripts')
    from patch_aware_context import summarize_patch as aware_summarize
    import tempfile, os

    test_cases = ["semantic", "interaction", "visual", "ui_flow", "copy_text", "other"]

    for change_class in test_cases:
        expected_grade = derive_grade(change_class).value

        # Check _load_patch_yaml
        with tempfile.NamedTemporaryFile(mode='w', suffix='.yaml', delete=False) as f:
            f.write(f'id: TEST\nchange_class: {change_class}\nstatus: draft\n')
            f.flush()
            loaded = _load_patch_yaml(f.name)
            assert loaded.get('grade_level') == expected_grade, \
                f"_load_patch_yaml: expected {expected_grade} for {change_class}, got {loaded.get('grade_level')}"
            os.unlink(f.name)

        # Check summarize_patch_for_context
        ctx_sum = summarize_patch_for_context({'id': 'x', 'change_class': change_class, 'scope': '', 'changed_files': [], 'status': 'approved'})
        assert expected_grade in ctx_sum, \
            f"summarize_patch_for_context: expected {expected_grade} in output for {change_class}"

        # Check patch_aware_context.py summarize_patch
        aware_sum = aware_summarize({'file_path': 'a.py', 'change_class': change_class, 'patch_status': 'pending'})
        assert aware_sum.get('grade_level') == expected_grade, \
            f"awareness summarize_patch: expected {expected_grade} for {change_class}, got {aware_sum.get('grade_level')}"
```

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| Patch YAML -> grade_level derivation | grade_level is deterministically derived from change_class via derive_grade() — no user input can override this |
| Context injection -> AI session | Context is informational only; the WARNING for Major patches guides AI behavior but does not enforce policy |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-10-10 | Integrity | derive_grade() in context injection | mitigate | grade_level is computed from change_class via deterministic mapping — not from user-supplied value |
| T-10-11 | Tampering | Patch YAML with forged grade_level | mitigate | _load_patch_yaml() re-derives grade_level from change_class if grade_level is missing; if present, it is trusted as the capture-time classification |
| T-10-12 | Spoofing | Major WARNING in context | accept | WARNING is informational, not an enforcement mechanism. Actual policy enforcement is in settle_runtime and frz_manage_runtime |
| T-10-16 | Integrity | Cross-caller derive_grade divergence | mitigate | All callers import derive_grade from patch_schema.py — cross-caller consistency test asserts same result |
</threat_model>
